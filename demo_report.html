Quick Start

git clone https://github.com/saigi/DataFew-shield.git
cd DataFew-shield
npm install
pip install -r requirements.txt

# Start services
python scripts/embedding_server.py --port 5000 &
node index.js --port 8080

# Run all benchmarks
bash demo.sh

# Or run individual tests
node test/test_gandalf_live.mjs
node test/test_owasp_top10.mjs
node test/test_mitre_atlas.mjs
node test/test_dlp.mjs
node test/test_adversarial.mjs
